/**
 * Détermine les blocs dans lequel l'étudiant est inscrit
 * @method
 * @param {string[]} groups Groupes de l'étudiant
 * @returns {boolean[]} 
 * @example
 *  determineBlocs(["1A", "2A"]); // [true, true, false]
 * @category Calendar
 */
const determineBlocs = (groups) => [isBlocAffiliate(groups, 1), isBlocAffiliate(groups, 2), isBlocAffiliate(groups, 3)];

async function routes(fastify, options){
    fastify.get('/calendar/bye', async (request, reply) => {
        coreCourses = new Set();
        courses = [];

        const currentCodes = getCurrentCodes(); // Codes API actuels
        const calendar = ical({name: "Cours - IESN Scheduler", timezone: "Europe/Brussels" }); // Initialisation du calendrier
        const validGroups = getValidBlocsAndGroupsBySection(); // Tous les groupes valides pour la section (ex: IG : 1a, 1b..., 2a, 2b, 3a, 3b);
        const requestedGroups = request.query["grp[]"]; // Les groupes indiqués dans la requête
        if(requestedGroups && requestedGroups.every(requestedGroup => validGroups.includes(requestedGroups))){
            [isBloc1, isBloc2, isBloc3] = determineBlocs(requestedGroups);
        }
        return { salut : "oui"};
    });
}

module.exports = routes;
